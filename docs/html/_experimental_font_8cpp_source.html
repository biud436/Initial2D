137 void ExperimentalFont::renderChar(HDC hdc, HBITMAP hBitmap, int xStart, int yStart, LPWSTR lpszText, COLORREF cr)
138 {
139  if (!m_bInit)
140  {
141  return;
142  }
143 
144  int x, y;
145  int width, height;
146  int line;
147 
148  BYTE alpha;
149  DWORD dwBufferSize;
150 
151  LPBYTE lp, lpBuffer;
152  GLYPHMETRICS gm;
153  MAT2 mat2 = {
154  {0,1}, {0,0},
155  {0,0}, {0,1}
156  };
157 
158  while (*lpszText)
159  {
160  dwBufferSize = GetGlyphOutlineW(hdc, *lpszText, GGO_GRAY8_BITMAP, &gm, 0, NULL, &mat2);
161  lpBuffer = new (std::nothrow) BYTE[dwBufferSize];
162  GetGlyphOutlineW(hdc, *lpszText, GGO_GRAY8_BITMAP, &gm, dwBufferSize, lpBuffer, &mat2);
163 
164  line = ((gm.gmBlackBoxX + 3) / 4) * 4;
165  width = gm.gmBlackBoxX;
166  height = gm.gmBlackBoxY;
167 
168  x = xStart + gm.gmptGlyphOrigin.x;
169  y = yStart - gm.gmptGlyphOrigin.y;
170 
171  if (x < 0) {
172  x = 0;
173  }
174 
175  if (y < 0) {
176  y = 0;
177  }
178 
179  for (int my = 0; my < height; my++)
180  {
181  for (int mx = 0; mx < width; mx++)
182  {
183  alpha = *(lpBuffer + (my * line) + mx);
184  if (alpha)
185  {
186  lp = getBits(hBitmap, x + mx, y + my);
187  lp[0] = (GetBValue(cr) * alpha / 64) + (lp[0] * (64 - alpha) / 64);
188  lp[1] = (GetGValue(cr) * alpha / 64) + (lp[1] * (64 - alpha) / 64);
189  lp[2] = (GetRValue(cr) * alpha / 64) + (lp[2] * (64 - alpha) / 64);
190  lp[3] = (255 * alpha / 64) + (lp[3] * (64 - alpha) / 64);
191  }
192  }
193  }
194 
195  // gmCellIncX는 다음 문자까지의 거리
196  xStart += gm.gmCellIncX;
197  lpszText++;
198 
199  delete[] lpBuffer;
200  }
201 }

205 {
