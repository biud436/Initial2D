98  int startFrame = m_spriteData.startFrame;
99  double delay = m_spriteData.frameDelay;
100 
101  if (endFrame - startFrame > 0)

109  m_nCurrentFrame++;
110 
111  if (m_nCurrentFrame < startFrame || m_nCurrentFrame > endFrame)
112  {

114  m_nCurrentFrame = startFrame;
115  } else
116  {
117  m_nCurrentFrame = endFrame;
118  m_bAnimComplete = true;

336 void Sprite::updateTransform()
