240 void App::Update()
241 {
242  // 경과된 시간(Elapsed)을 구한다.
243  QueryPerformanceCounter(&m_nTimeEnd);
244 
245  m_frameTime = (double)(m_nTimeEnd.QuadPart - m_nTimeStart.QuadPart) / (double)m_nTimeFreq.QuadPart;
246 
247  m_elapsed += m_frameTime;
248 
249  m_nFrameCount++;
250 
251  // 부동 소수점 오차를 줄이기 위해 정밀도가 높은 double 형을 사용하였다.
252  while(m_elapsed >= DELAY_TIME)
253  {
254  UpdateInput();
255  ObjectUpdate(m_frameTime);
256  m_elapsed -= DELAY_TIME;
257  Sleep(1);
258  }
259 
260  if (m_frameTime >= 1.0) {
261  std::ostringstream oss;
262  oss << GAME_TITLE << m_nFrameCount << std::endl;
263  SetWindowText(m_hWnd, oss.str().c_str());
264  m_nFPS = m_nFrameCount;
265  m_nFrameCount = 0;
266  m_elapsed = 0.0;
267  m_nTimeStart = m_nTimeEnd;
268  }
269 
270  // Rendering
271  RenderClear();
272  RenderTransform();
273  Render();
274  RenderPresent();
275 
276 }
