120 TextureData* LoadBMP(std::string fileName)
121 {
122  HANDLE hFile;
123  DWORD dwFileSize, dwRead;
124  BITMAPFILEHEADER bhFileHeader;
125  BITMAPINFO *ih;
126  PVOID pRaster;
127  TextureData *pTextureData = new TextureData();
128 
129  hFile = CreateFile(fileName.c_str(), GENERIC_READ, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
130  if (hFile == INVALID_HANDLE_VALUE) {
131  MessageBox(g_hWnd, "파일을 찾을 수 없습니다", "오류", MB_OK | MB_ICONERROR);
132  return pTextureData;
133  }
134 
135  // 파일 헤더와 정보 구조체 취득
136  ReadFile(hFile, &bhFileHeader, sizeof(BITMAPFILEHEADER), &dwRead, NULL);
137  dwFileSize = bhFileHeader.bfOffBits - sizeof(BITMAPFILEHEADER);
138  ih = (BITMAPINFO *)malloc(dwFileSize);
139  ReadFile(hFile, ih, dwFileSize, &dwRead, NULL);
140 
141  // DIB 섹션 및 버퍼 할당
142  pTextureData->texture = CreateDIBSection(NULL, ih, DIB_RGB_COLORS, &pRaster, NULL, 0);
143 
144  // 래스터 데이터 취득
145  ReadFile(hFile, pRaster, bhFileHeader.bfSize - bhFileHeader.bfOffBits, &dwRead, NULL);
146  free(ih);
147  ih = NULL;
148  CloseHandle(hFile);
149 
150  pTextureData->width = ih->bmiHeader.biWidth;
151  pTextureData->height = ih->bmiHeader.biHeight;
152 
153  return pTextureData;
154

157 TextureData* LoadPNG(std::string fileName)
158 {
159 
160  TextureData *pTextureData = new TextureData();
161 
162  int width, height;
163  png_byte color_type;
164  png_byte bit_depth;
165  png_bytep* row_pointers;
166 
167  VOID *pvBits;
168 
169  BITMAPINFOHEADER bmih;
170 
171  // 파일을 불러옵니다
172  FILE *fp = fopen(fileName.c_str(), "rb");
173 
174  if (fp == NULL)
175  {
176  MessageBox(g_hWnd, "이미지 파일을 찾을 수 없습니다", "오류", MB_OK | MB_ICONERROR);
177  CLOSE_PNG_FILE(fp, pTextureData,"파일을 불러오는데 실패했습니다")
178  return pTextureData;
179  }
180 
181  // PNG 파일을 읽을 구조체를 생성합니다.
182  png_structp png_ptr = g_png_func.png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
183 
184  if (png_ptr == NULL) {
185  CLOSE_PNG_FILE(fp, pTextureData, "png_ptr 실패")
186  return pTextureData;
187  }
188 
189  png_infop info_ptr = g_png_func.png_create_info_struct(png_ptr);
190 
191  if (info_ptr == NULL) {
192  CLOSE_PNG_FILE(fp, pTextureData, "info_ptr 실패")
193  g_png_func.png_destroy_read_struct(&png_ptr, NULL, NULL);
194  return pTextureData;
195  }
196 
197 #ifdef PNG_DYNAMIC_LIBRARY
198  if (setjmp( *g_png_func.png_set_longjmp_fn(png_ptr, longjmp, sizeof(jmp_buf)) ))
199 #else
200  if (setjmp(png_jmpbuf(png_ptr)))
201 #endif
202  {
203  MessageBox(NULL, "setjmp 실패", "", MB_OK);
204  CLOSE_PNG_FILE(fp, pTextureData, "setjmp 실패")
205  g_png_func.png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
206  return pTextureData;
207  }
208 
209  g_png_func.png_init_io(png_ptr, fp);
210  g_png_func.png_read_info(png_ptr, info_ptr);
211 
212  // 이미지의 폭
213  width = g_png_func.png_get_image_width(png_ptr, info_ptr);
214 
215  // 이미지의 높이
216  height = g_png_func.png_get_image_height(png_ptr, info_ptr);
217 
218  // 컬러 타입
219  color_type = g_png_func.png_get_color_type(png_ptr, info_ptr);
220 
221  // 비트 깊이
222  bit_depth = g_png_func.png_get_bit_depth(png_ptr, info_ptr);
223 
224  if (bit_depth == 16)
225  {
226  g_png_func.png_set_strip_16(png_ptr);
227  }
228 
229  if (color_type == PNG_COLOR_TYPE_PALETTE)
230  {
231  g_png_func.png_set_palette_to_rgb(png_ptr);
232  }
233 
234  if (color_type == PNG_COLOR_TYPE_GRAY && bit_depth < 8)
235  {
236  g_png_func.png_set_expand_gray_1_2_4_to_8(png_ptr);
237  }
238 
239  // 알파값 처리
240  if (g_png_func.png_get_valid(png_ptr, info_ptr, PNG_INFO_tRNS))
241  {
242  g_png_func.png_set_tRNS_to_alpha(png_ptr);
243  }
244 
245  if (color_type == PNG_COLOR_TYPE_RGB ||
246  color_type == PNG_COLOR_TYPE_GRAY ||
247  color_type == PNG_COLOR_TYPE_PALETTE)
248  {
249  g_png_func.png_set_filler(png_ptr, 0xFF, PNG_FILLER_AFTER);
250  }
251 
252  if (color_type == PNG_COLOR_TYPE_GRAY ||
253  color_type == PNG_COLOR_TYPE_GRAY_ALPHA)
254  {
255  g_png_func.png_set_gray_to_rgb(png_ptr);
256  }
257 
258  if (bit_depth < 8)
259  {
260  g_png_func.png_set_packing(png_ptr);
261  }
262 
263  g_png_func.png_read_update_info(png_ptr, info_ptr);
264 
265  // 비트 깊이
266  bit_depth = g_png_func.png_get_bit_depth(png_ptr, info_ptr);
267 
268  // 색상 타입
269  color_type = g_png_func.png_get_color_type(png_ptr, info_ptr);
270 
271  // 비트맵 스캔 라인을 위한 메모리 할당
272  row_pointers = (png_bytep*)malloc(sizeof(png_bytep) * height);
273 
274  for (int y = 0; y < height; y++)
275  {
276  row_pointers[y] = (png_byte*)malloc(g_png_func.png_get_rowbytes(png_ptr, info_ptr));
277  }
278 
279  g_png_func.png_read_image(png_ptr, row_pointers);
280  g_png_func.png_read_end(png_ptr, NULL);
281  g_png_func.png_destroy_read_struct(&png_ptr, &info_ptr, 0);
282 
283  fclose(fp);
284 
285  memset(&bmih, 0, sizeof(BITMAPINFOHEADER));
286 
287  // 비트맵 구조체 정의
288  bmih.biWidth = width;
289  bmih.biHeight = -height; // 비트맵은 상하 반전이므로 음수 값으로 반전 처리
290  bmih.biBitCount = 32; // 32비트
291  bmih.biCompression = BI_RGB;
292  bmih.biSize = sizeof(BITMAPINFOHEADER);
293  bmih.biSizeImage = width * height * 4;
294  bmih.biPlanes = 1;
295  bmih.biXPelsPerMeter = 2835; // 72 DPI × 39.3701 인치, 2834.6472
296  bmih.biYPelsPerMeter = 2835;
297 
298  BITMAPINFO* bmi = (BITMAPINFO*)&bmih;
299 
300  // DIB 영역을 만든다.
301  pTextureData->width = width;
302  pTextureData->height = height;
303  pTextureData->texture = CreateDIBSection(App::GetInstance().GetContext().mainContext, bmi, DIB_RGB_COLORS, &pvBits, NULL, 0x0);
304 
305  // 비트맵을 색상 값으로 채운다.
306  // @link MSDN - https://msdn.microsoft.com/ko-kr/library/windows/desktop/dd183353(v=vs.85).aspx
307  for (int y = 0; y < height; y++)
308  {
309 
310  png_bytep row = row_pointers[y];
311 
312  for (int x = 0; x < width; x++)
313  {
314  png_bytep px = &(row[x * 4]); // 4-Byte Alignments
315 
316  // 4바이트라면,
317  if ((x * 4) % 4 == 0) {
318 
319  // 비트 플래그로 RGBA -> ARGB로 변환한다.
320  ((UINT32 *)pvBits)[y * width + x] = (px[3]<< 24)|(px[0]<<16)|(px[1]<<8)|(px[2]);
321 
322  }
323  }
324 
325  }
326 
327  // 메모리 정리
328  free(row_pointers);
329 
330  row_pointers = NULL;
331 
332  // 비트맵 반환
333  return pTextureData;
334 
335 }
